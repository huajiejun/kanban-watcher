# 差异文件模块设计

## 概述

在 kanban-watcher 中实现差异文件功能，用于实时追踪和展示工作区的文件变更。

## vibe-kanban 接口分析

### 1. WebSocket 差异流接口

**端点**: `GET /api/workspaces/{workspace_id}/git/diff/ws`

**查询参数**:
- `stats_only`: 是否只返回统计信息
- `path_prefix`: 路径前缀过滤

**消息格式** (LogMsg):
```json
{
  "type": "json_patch",
  "patch": {
    "op": "add" | "remove",
    "path": "/diffs/{file_path}",
    "value": { ...Diff }
  }
}
```

### 2. 分支状态接口

**端点**: `GET /api/workspaces/{workspace_id}/git/status`

**响应**:
```json
{
  "success": true,
  "data": [{
    "repo_id": "uuid",
    "repo_name": "string",
    "status": {
      "commits_ahead": 0,
      "commits_behind": 0,
      "has_uncommitted_changes": true,
      "uncommitted_count": 5,
      "untracked_count": 2,
      "target_branch_name": "main"
    }
  }]
}
```

### 数据类型 (types.go)

```go
package diff

import "time"

// DiffKind 变更类型
type DiffKind string

const (
    DiffAdded    DiffKind = "added"
    DiffModified DiffKind = "modified"
    DiffDeleted  DiffKind = "deleted"
    DiffRenamed  DiffKind = "renamed"
)

// Diff 文件差异
type Diff struct {
    Change         DiffKind `json:"change"`
    OldPath        *string  `json:"old_path,omitempty"`
    NewPath        *string  `json:"new_path,omitempty"`
    OldContent     *string  `json:"old_content,omitempty"`
    NewContent     *string  `json:"new_content,omitempty"`
    Additions      *int     `json:"additions,omitempty"`
    Deletions      *int     `json:"deletions,omitempty"`
    ContentOmitted bool     `json:"content_omitted"`
    RepoID         *string  `json:"repo_id,omitempty"`
}

// DiffStats 差异统计
type DiffStats struct {
    FilesChanged int `json:"files_changed"`
    LinesAdded   int `json:"lines_added"`
    LinesRemoved int `json:"lines_removed"`
}

// WorkspaceDiff 工作区差异汇总
type WorkspaceDiff struct {
    WorkspaceID string          `json:"workspace_id"`
    UpdatedAt   time.Time       `json:"updated_at"`
    Diffs       map[string]Diff `json:"diffs"`
    Stats       DiffStats       `json:"stats"`
}

// PatchMessage WebSocket 补丁消息
type PatchMessage struct {
    Op    string `json:"op"`
    Path  string `json:"path"`
    Value *Diff  `json:"value,omitempty"`
}

// LogMsg 日志消息
type LogMessage struct {
    Type    string        `json:"type"`
    Patch   *PatchMessage `json:"patch,omitempty"`
    Ready   bool          `json:"ready,omitempty"`
}
```

### 客户端 (client.go)

```go
package diff

import (
    "context"
    "encoding/json"
    "fmt"
    "net/http"
    "sync"
    "time"

    "gorilla/websocket"
)

// Client 差异流客户端
type Client struct {
    baseURL    string
    httpClient *http.Client
    wsDialer   *websocket.Dialer

    mu       sync.RWMutex
    streams  map[string]*Stream // workspace_id -> Stream
    onDiff   func(workspaceID string, diff *WorkspaceDiff)
}

// Stream 差异流
type Stream struct {
    WorkspaceID string
    conn        *websocket.Conn
    diff        *WorkspaceDiff
    done        chan struct{}
}

// NewClient 创建客户端
func NewClient(baseURL string) *Client {
    return &Client{
        baseURL:    baseURL,
        httpClient: &http.Client{Timeout: 30 * time.Second},
        wsDialer:   websocket.DefaultDialer,
        streams:    make(map[string]*Stream),
    }
}

// OnDiff 设置差异回调
func (c *Client) OnDiff(handler func(workspaceID string, diff *WorkspaceDiff)) {
    c.onDiff = handler
}

// Subscribe 订阅工作区差异流
func (c *Client) Subscribe(ctx context.Context, workspaceID, token string) error {
    c.mu.Lock()
    defer c.mu.Unlock()

    if _, exists := c.streams[workspaceID]; exists {
        return nil // 已订阅
    }

    url := fmt.Sprintf("%s/api/workspaces/%s/git/diff/ws", c.baseURL, workspaceID)

    header := http.Header{}
    header.Set("Authorization", "Bearer "+token)

    conn, _, err := c.wsDialer.Dial(url, header)
    if err != nil {
        return fmt.Errorf("连接差异流失败: %w", err)
    }

    stream := &Stream{
        WorkspaceID: workspaceID,
        conn:        conn,
        diff: &WorkspaceDiff{
            WorkspaceID: workspaceID,
            Diffs:       make(map[string]Diff),
            UpdatedAt:   time.Now(),
        },
        done: make(chan struct{}),
    }

    c.streams[workspaceID] = stream

    go c.handleStream(ctx, stream)

    return nil
}

// handleStream 处理差异流消息
func (c *Client) handleStream(ctx context.Context, stream *Stream) {
    defer func() {
        stream.conn.Close()
        c.mu.Lock()
        delete(c.streams, stream.WorkspaceID)
        c.mu.Unlock()
    }()

    for {
        select {
        case <-ctx.Done():
            return
        case <-stream.done:
            return
        default:
            _, message, err := stream.conn.ReadMessage()
            if err != nil {
                return
            }

            var msg LogMessage
            if err := json.Unmarshal(message, &msg); err != nil {
                continue
            }

            c.processMessage(stream, &msg)
        }
    }
}

// processMessage 处理消息
func (c *Client) processMessage(stream *Stream, msg *LogMessage) {
    if msg.Type != "json_patch" || msg.Patch == nil {
        return
    }

    patch := msg.Patch
    filePath := patch.Path // "/diffs/path/to/file"

    stream.mu.Lock()
    defer stream.mu.Unlock()

    switch patch.Op {
    case "add":
        if patch.Value != nil {
            stream.diff.Diffs[filePath] = *patch.Value
            stream.diff.UpdatedAt = time.Now()
        }
    case "remove":
        delete(stream.diff.Diffs, filePath)
        stream.diff.UpdatedAt = time.Now()
    }

    // 重新计算统计
    stream.diff.Stats = calculateStats(stream.diff.Diffs)

    // 触发回调
    if c.onDiff != nil {
        c.onDiff(stream.WorkspaceID, stream.diff)
    }
}

// Unsubscribe 取消订阅
func (c *Client) Unsubscribe(workspaceID string) {
    c.mu.Lock()
    defer c.mu.Unlock()

    if stream, exists := c.streams[workspaceID]; exists {
        close(stream.done)
    }
}

// GetDiff 获取当前差异
func (c *Client) GetDiff(workspaceID string) *WorkspaceDiff {
    c.mu.RLock()
    defer c.mu.RUnlock()

    if stream, exists := c.streams[workspaceID]; exists {
        stream.mu.RLock()
        defer stream.mu.RUnlock()
        return stream.diff
    }
    return nil
}

func calculateStats(diffs map[string]Diff) DiffStats {
    stats := DiffStats{}
    for _, d := range diffs {
        stats.FilesChanged++
        if d.Additions != nil {
            stats.LinesAdded += *d.Additions
        }
        if d.Deletions != nil {
            stats.LinesRemoved += *d.Deletions
        }
    }
    return stats
}
```

#### 2. 在 Workspace 菜单中展示差异统计

```go
// internal/tray/workspace_menu_formatter.go

func (f *Formatter) formatDiffStats(d *diff.WorkspaceDiff) string {
    if d == nil || d.Stats.FilesChanged == 0 {
        return ""
    }

    return fmt.Sprintf(" [%d 文件变更, +%d/-%d]",
        d.Stats.FilesChanged,
        d.Stats.LinesAdded,
        d.Stats.LinesRemoved)
}
```

## 使用示例

```go
// 订阅差异流
client := diff.NewClient("https://vibe-kanban.example.com")
client.OnDiff(func(workspaceID string, d *diff.WorkspaceDiff) {
    log.Printf("工作区 %s 有 %d 个文件变更", workspaceID, d.Stats.FilesChanged)
})

err := client.Subscribe(ctx, workspaceID, authToken)
if err != nil {
    log.Fatal(err)
}

// 获取当前差异
currentDiff := client.GetDiff(workspaceID)
for path, d := range currentDiff.Diffs {
    fmt.Printf("%s: %s\n", path, d.Change)
}
```

## 注意事项

1. **连接管理**: 处理断线重连逻辑
2. **内存控制**: 大文件差异可能占用大量内存，需要设置上限
3. **并发安全**: 使用互斥锁保护共享状态
4. **错误处理**: WebSocket 连接异常时的优雅降级
